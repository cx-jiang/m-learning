def matxMultiply(A, B):
    new_row,A_col = shape(A)
    B_row,new_col = shape(B)
    if A_col != B_row:
        raise ValueError
    N = [[0]*new_col for row in range(new_row)]
    for i in range(0,new_row):
        for j in range(0,A_col):
            for k in range(0,new_col):
                N[i][k] += A[i][j] * B[j][k]
    return N

""" Gaussian Jordan  Ax = b.
    
        A:  
        b: 
        decPts: 4
        epsilon: 0 1.0e-16
        
     x  Ax = b 
    None Ab 
    None A 
"""
def gj_Solve(A, b, decPts=4, epsilon=1.0e-16):
    a_row,a_col = shape(A)
    b_row,b_col = shape(b)
    if a_row != b_row:
        return None
    
    Ab = augmentMatrix(A,b)
    
    for i in range(0,a_col):
        # 
        max_value_row = i
        for j in range(i, a_row):
            if abs(Ab[j][i]) > abs(Ab[max_value_row][i]):
                max_value_row = j
        if i != max_value_row:
            swapRows(Ab, i, max_value_row)
            
        # 1
        if abs(Ab[i][i]) < epsilon:
            return None
        else:
            scaleRow(Ab,i,Fraction(1/Ab[i][i]))

        # 0
        for j in range(0,a_row):
            if i == j:
                continue
            else:
                addScaledRow(Ab,j,i,-Ab[j][i])
    # 
    matxRound(Ab,4)
    return [row[-1:] for row in Ab]
